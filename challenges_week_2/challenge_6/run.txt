# create marine company
company = marine_company("oceanic_transports")

# create ships
cruise1 = cruise_ship(1, "sea_queen", "luxury")
cargo1 = cargo_ship(2, "cargo_master", 5000)

company.add_ship(cruise1)
company.add_ship(cargo1)

# create customers
c1 = customer(1, "arjun", 5000, "goa", "mumbai")
c2 = customer(2, "meera", 4500, "goa", "kochi")
c3 = customer(3, "logistics_inc", 12000, "chennai", "dubai")

# register customers
cruise1.add_customer(c1)
cruise1.add_customer(c2)
cargo1.add_customer(c3)

# outputs
print("total amount collected:", company.total_amount_collected())

print("\ntotal amount per ship:")
for ship_name, amount in company.total_amount_per_ship().items():
    print(ship_name, ":", amount)

print("\ncustomers for cruise ship sea_queen:")
for cust in cruise1.get_customer_details():
    print(cust.name, cust.travel_from, "->", cust.travel_to)
